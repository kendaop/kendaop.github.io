par(cex=2, pch=20, oma=c(1,1,0,0))
data = data.frame(x=c(1:4), y=c((1:4)^2))
fit = lm(y~x, data=data)
plot(data$y ~ data$x, col="blue", xlab="X", ylab="Y")
abline(fit, col="red", lwd=2)
